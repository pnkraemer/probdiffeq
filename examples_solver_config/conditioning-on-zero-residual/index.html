    Probabilistic solvers as collocation methods¶

def log_residual(*args):
    """Evaluate the log-ODE-residual."""
    return jnp.log10(jnp.abs(residual(*args)))


def residual(x, t):
    """Evaluate the ODE residual."""
    return x[..., 1] - jax.vmap(vector_field)(x[..., 0], t)
